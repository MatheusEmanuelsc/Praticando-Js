1. 🌟 Introdução à Manipulação do DOM

A manipulação de elementos no DOM é fundamental para criar páginas web dinâmicas e interativas. JavaScript oferece uma ampla gama de métodos para criar, remover, modificar e gerenciar elementos HTML.

2. ✨ Criando Elementos

JavaScript oferece várias formas de criar novos elementos HTML dinamicamente.

// Método básico: createElement const novoDiv = document.createElement('div'); novoDiv.textContent = 'Novo elemento criado!'; novoDiv.className = 'minha-classe'; novoDiv.id = 'meu-id'; // Adicionando ao DOM document.body.appendChild(novoDiv); // Criando elementos com atributos const novaImagem = document.createElement('img'); novaImagem.src = 'imagem.jpg'; novaImagem.alt = 'Descrição da imagem'; novaImagem.style.width = '200px'; // Criando elementos mais complexos function criarCard(titulo, conteudo, imagemSrc) { const card = document.createElement('div'); card.className = 'card'; const imagem = document.createElement('img'); imagem.src = imagemSrc; imagem.alt = titulo; const tituloEl = document.createElement('h3'); tituloEl.textContent = titulo; const conteudoEl = document.createElement('p'); conteudoEl.textContent = conteudo; const botao = document.createElement('button'); botao.textContent = 'Ver mais'; botao.onclick = () => alert(`Clicou em: ${titulo}`); card.appendChild(imagem); card.appendChild(tituloEl); card.appendChild(conteudoEl); card.appendChild(botao); return card; } // Criando múltiplos elementos function criarLista(itens) { const lista = document.createElement('ul'); itens.forEach(item => { const li = document.createElement('li'); li.textContent = item; li.addEventListener('click', () => { li.style.textDecoration = 'line-through'; }); lista.appendChild(li); }); return lista; } // Usando DocumentFragment para performance function criarMuitosElementos(quantidade) { const fragment = document.createDocumentFragment(); for (let i = 0; i < quantidade; i++) { const div=document.createElement('div'); div.textContent=`Elemento ${i + 1}`; div.className='item-lista' ; fragment.appendChild(div); } document.getElementById('container').appendChild(fragment); } // Criando com template strings (innerHTML) function criarElementoHTML(tag, conteudo, classes = '') { const elemento = document.createElement(tag); if (classes) elemento.className = classes; elemento.innerHTML = conteudo; return elemento; }

3. 🗑️ Removendo Elementos

Existem várias maneiras de remover elementos do DOM, cada uma com suas particularidades.

// Método moderno: remove() const elemento = document.getElementById('meuElemento'); elemento.remove(); // Remove o elemento diretamente // Método clássico: removeChild() const pai = document.getElementById('container'); const filho = document.getElementById('item1'); pai.removeChild(filho); // Removendo todos os filhos function limparContainer(container) { while (container.firstChild) { container.removeChild(container.firstChild); } // Ou mais simples: // container.innerHTML = ''; } // Removendo elementos por seletor function removerPorClasse(classe) { const elementos = document.querySelectorAll(`.${classe}`); elementos.forEach(el => el.remove()); } // Remover com confirmação function removerComConfirmacao(elemento) { if (confirm('Tem certeza que deseja remover este elemento?')) { elemento.remove(); console.log('Elemento removido!'); } } // Remover com animação function removerComAnimacao(elemento) { elemento.style.transition = 'opacity 0.5s, transform 0.5s'; elemento.style.opacity = '0'; elemento.style.transform = 'scale(0.8)'; setTimeout(() => { elemento.remove(); }, 500); } // Remover múltiplos elementos function removerSelecionados() { const selecionados = document.querySelectorAll('.selecionado'); selecionados.forEach(elemento => { removerComAnimacao(elemento); }); } // Remover e manter referência function removerTemporariamente(elemento) { const pai = elemento.parentNode; const proximo = elemento.nextSibling; pai.removeChild(elemento); // Função para restaurar return function restaurar() { pai.insertBefore(elemento, proximo); }; }

4. 👁️ Esconder e Mostrar Elementos

Diferentes técnicas para controlar a visibilidade de elementos sem removê-los do DOM.

// Usando display (remove do layout) function esconder(elemento) { elemento.style.display = 'none'; } function mostrar(elemento, tipoDisplay = 'block') { elemento.style.display = tipoDisplay; } // Usando visibility (mantém espaço no layout) function tornarInvisivel(elemento) { elemento.style.visibility = 'hidden'; } function tornarVisivel(elemento) { elemento.style.visibility = 'visible'; } // Usando opacity (transparência) function tornarTransparente(elemento) { elemento.style.opacity = '0'; } function tornarOpaco(elemento) { elemento.style.opacity = '1'; } // Toggle com classes CSS function alternarVisibilidade(elemento) { elemento.classList.toggle('hidden'); } // Esconder com animação suave function esconderComAnimacao(elemento) { elemento.style.transition = 'opacity 0.3s, transform 0.3s'; elemento.style.opacity = '0'; elemento.style.transform = 'translateY(-20px)'; setTimeout(() => { elemento.style.display = 'none'; }, 300); } function mostrarComAnimacao(elemento) { elemento.style.display = 'block'; elemento.style.opacity = '0'; elemento.style.transform = 'translateY(-20px)'; setTimeout(() => { elemento.style.opacity = '1'; elemento.style.transform = 'translateY(0)'; }, 10); } // Fade in/out function fadeOut(elemento, duracao = 300) { const inicio = Date.now(); const opacidadeInicial = parseFloat(getComputedStyle(elemento).opacity); function animar() { const progresso = (Date.now() - inicio) / duracao; if (progresso < 1) { elemento.style.opacity=opacidadeInicial * (1 - progresso); requestAnimationFrame(animar); } else { elemento.style.opacity='0' ; elemento.style.display='none' ; } } animar(); } function fadeIn(elemento, duracao=300) { elemento.style.display='block' ; elemento.style.opacity='0' ; const inicio=Date.now(); function animar() { const progresso=(Date.now() - inicio) / duracao; if (progresso < 1) { elemento.style.opacity=progresso; requestAnimationFrame(animar); } else { elemento.style.opacity='1' ; } } animar(); } // Slide up/down function slideUp(elemento, duracao = 300) { const altura = elemento.offsetHeight; elemento.style.transition = `height ${duracao}ms`; elemento.style.overflow = 'hidden'; elemento.style.height = altura + 'px'; setTimeout(() => { elemento.style.height = '0px'; setTimeout(() => { elemento.style.display = 'none'; }, duracao); }, 10); } function slideDown(elemento, duracao = 300) { elemento.style.display = 'block'; const altura = elemento.scrollHeight; elemento.style.height = '0px'; elemento.style.overflow = 'hidden'; elemento.style.transition = `height ${duracao}ms`; setTimeout(() => { elemento.style.height = altura + 'px'; setTimeout(() => { elemento.style.height = 'auto'; elemento.style.overflow = 'visible'; }, duracao); }, 10); }

5. 📋 Clonando Elementos

Clonar elementos permite duplicar elementos existentes no DOM, com ou sem seus eventos.

// Clonagem superficial (sem filhos) const elemento = document.getElementById('item1'); const cloneSuperficial = elemento.cloneNode(false); // Clonagem profunda (com filhos) const cloneProfundo = elemento.cloneNode(true); // Clonagem com modificações function clonarEModificar(elemento, novoTexto) { const clone = elemento.cloneNode(true); clone.textContent = novoTexto; clone.id = `clone-${Math.random().toString(36).substr(2, 9)}`; return clone; } // Clonagem com eventos function clonarComEventos(elemento) { const clone = elemento.cloneNode(true); clone.addEventListener('click', () => { alert('Clone clicado!'); }); return clone; } // Clonagem múltipla function clonarVariasVezes(elemento, quantidade) { const fragment = document.createDocumentFragment(); for (let i = 0; i < quantidade; i++) { const clone=elemento.cloneNode(true); clone.textContent=`${clone.textContent} (${i + 1})`; fragment.appendChild(clone); } return fragment; }

6. 🚚 Movendo Elementos

Mover elementos dentro do DOM é útil para reorganizar a estrutura da página dinamicamente.

// Mover com appendChild function moverParaContainer(elemento, container) { container.appendChild(elemento); } // Mover com insertBefore function moverAntes(elemento, referencia) { referencia.parentNode.insertBefore(elemento, referencia); } // Mover para o início function moverParaInicio(elemento, container) { container.insertBefore(elemento, container.firstChild); } // Mover com animação function moverComAnimacao(elemento, destino) { const rect = elemento.getBoundingClientRect(); elemento.style.transition = 'transform 0.5s'; elemento.style.transform = `translate(${destino.getBoundingClientRect().x - rect.x}px, ${destino.getBoundingClientRect().y - rect.y}px)`; setTimeout(() => { elemento.style.transition = ''; elemento.style.transform = ''; destino.appendChild(elemento); }, 500); }

7. 📥 Inserindo Elementos

Inserir elementos em posições específicas do DOM é essencial para layouts dinâmicos.

// Inserir no final function inserirNoFinal(elemento, container) { container.appendChild(elemento); } // Inserir no início function inserirNoInicio(elemento, container) { container.insertBefore(elemento, container.firstChild); } // Inserir antes de um elemento function inserirAntes(elemento, referencia) { referencia.parentNode.insertBefore(elemento, referencia); } // Inserir usando insertAdjacentElement function inserirAdjacente(elemento, posicao, referencia) { referencia.insertAdjacentElement(posicao, elemento); // Posições: 'beforebegin', 'afterbegin', 'beforeend', 'afterend' }

8. 🔄 Substituindo Elementos

Substituir elementos permite trocar um elemento por outro no DOM.

// Substituir com replaceChild function substituirElemento(novo, antigo) { antigo.parentNode.replaceChild(novo, antigo); } // Substituir com replaceWith function substituirComReplaceWith(elemento, novo) { elemento.replaceWith(novo); } // Substituir múltiplos function substituirSelecionados(classe, novoElemento) { const elementos = document.querySelectorAll(`.${classe}`); elementos.forEach(el => { const clone = novoElemento.cloneNode(true); el.replaceWith(clone); }); }

9. 🎨 Manipulando Atributos e Classes

Controlar atributos e classes é essencial para interatividade e estilização dinâmica.

// Manipulando atributos function definirAtributo(elemento, atributo, valor) { elemento.setAttribute(atributo, valor); } function obterAtributo(elemento, atributo) { return elemento.getAttribute(atributo); } function removerAtributo(elemento, atributo) { elemento.removeAttribute(atributo); } // Manipulando classes function adicionarClasse(elemento, classe) { elemento.classList.add(classe); } function removerClasse(elemento, classe) { elemento.classList.remove(classe); } function alternarClasse(elemento, classe) { elemento.classList.toggle(classe); } // Verificando classes function temClasse(elemento, classe) { return elemento.classList.contains(classe); }

10. 🎨 Manipulando Estilos

Modificar estilos diretamente via JavaScript permite personalização dinâmica.

// Alterando estilos inline function definirEstilo(elemento, propriedade, valor) { elemento.style[propriedade] = valor; } // Obtendo estilos computados function obterEstilo(elemento, propriedade) { return getComputedStyle(elemento)[propriedade]; } // Alternando estilos function alternarEstilo(elemento, propriedade, valor1, valor2) { elemento.style[propriedade] = elemento.style[propriedade] === valor1 ? valor2 : valor1; } // Aplicando múltiplos estilos function aplicarEstilos(elemento, estilos) { Object.assign(elemento.style, estilos); }

11. 📊 Comparação de Métodos

Comparação dos principais métodos de manipulação do DOM.

Método Descrição Vantagens Limitações
createElement Cria um novo elemento HTML Seguro, flexível Requer configuração manual
remove Remove elemento diretamente Simples, moderno Não suporta IE
appendChild Adiciona elemento como último filho Amplamente suportado Limitado a um elemento
insertBefore Insere antes de um elemento Precisão na inserção Requer referência
replaceWith Substitui elemento Simples, intuitivo Suporte recente

12. ✅ Boas Práticas

  • Evite innerHTML: Use métodos como createElement para segurança contra XSS.
  • Use DocumentFragment: Melhora performance ao adicionar múltiplos elementos.
  • Cache seletores: Armazene resultados de querySelector em variáveis.
  • Manipule classes: Prefira classList em vez de estilos inline.
  • Gerencie eventos: Remova event listeners quando elementos são removidos.
  • Teste compatibilidade: Verifique suporte para métodos modernos como replaceWith.
💡 Dica Final: Sempre valide entradas do usuário e otimize manipulações do DOM para melhor performance.
